173 inline std::pair<UnitTest::helper::Case, UnitTest::helper::Case>
174 UnitTest::helper::MS_loops(const Wavefunction &wf,
175  const DiracOperator::TensorOperator *h) {
176  // compare the best and worst! (of each operator)
177  Case best{999.0};
178  Case worst{0.0};
179 
180  const auto omega_mults = std::vector{0.0, 0.25};
181 
182 #pragma omp parallel for
183  for (auto i = 0ul; i < wf.valence.size(); ++i) {
184  const auto &Fv = wf.valence[i];
185  const auto vl = wf.get_Vlocal(Fv.l());
186  for (const auto &Fm : wf.valence) {
187 
188  // Only do for cases that make sense:
189  if (Fm == Fv)
190  continue; // gives 0/0
191  if (h->isZero(Fv.k, Fm.k))
192  continue;
193  if (Fm.k != Fv.k && std::abs(Fm.n - Fv.n) != 0)
194  continue;
195 
196  const auto h_mv = h->reducedME(Fm, Fv);
197 
198  // Don't do in cases where ME is extremely small. NOTE: This requires
199  // a good choice of units, since ME may be small due to small
200  // coeficient, which should not count as 'small'
201  if (std::abs(h_mv) < 1.0e-8) {
202  continue;
203  }
204 
205  // Form 'rhs': (h - de_A)|A>
206  auto hFv = h->reduced_rhs(Fm.k, Fv);
207  if (Fm.k == Fv.k) {
208  auto de = h->reducedME(Fv, Fv);
209  hFv -= de * Fv;
210  }
211 
212  // loop over few frequencies:
213  for (const auto &w_mult : omega_mults) {
214  const auto w = std::abs(Fv.en * w_mult);
215 
216  const auto dFv = ExternalField::solveMixedState(Fm.k, Fv, w, vl,
217  wf.alpha, wf.core, hFv);
218 
219  const auto lhs = Fm * dFv;
220  const auto rhs = h_mv / (Fv.en - Fm.en + w);
221  const auto eps = (lhs - rhs) / (lhs + rhs);
222 
223 // find the best and worst case:
224 #pragma omp critical(find_max)
225  {
226  if (std::abs(eps) > std::abs(worst.eps)) {
227  worst.eps = eps;
228  worst.name = Fm.shortSymbol() + "|" + Fv.shortSymbol();
229  worst.w = w;
230  }
231  if (std::abs(eps) < std::abs(best.eps)) {
232  best.eps = eps;
233  best.name = Fm.shortSymbol() + "|" + Fv.shortSymbol();
234  best.w = w;
235  }
236  }
237  }
238  }
239  }
240  return {best, worst};
241 }
