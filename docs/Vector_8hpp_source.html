226  const auto interval = static_cast<double>(last - first);
227  range.push_back(first); // guarentee first is first
228  for (N i = 1; i < number - 1; ++i) {

230  const auto value = static_cast<double>(first) + (eps * interval);
231  range.push_back(static_cast<T>(value));
232  }
