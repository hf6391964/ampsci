27 template <typename C, typename... Args>
28 inline double integrate(const double dt, std::size_t beg, std::size_t end,
29  const C &f1, const Args &... rest)
30 // // Note: includes no safety checks!
31 // // Integrates from (point) beg to end-1 (i.e., not including end)
32 // // Require:
33 // // * (beg-end) > 2*Nquad
34 // // * end - Nquad > Nquad
35 // // * beg + 2*Nquad
36 // // NB: end-point corrections only applied if beg < Nquad, end > max - nquad
37 // // Not sure if this is best choice - but it ensures that:
38 // // int_{a->b} + int_{b->c} = int_{a->c}
39 {
40  [[maybe_unused]] auto sp = IO::Profile::safeProfiler(__func__, "4");
41 
42  const auto max_grid = f1.size();
43  if (end == 0)
44  end = max_grid;
45  const auto end_mid = std::min(max_grid - Nquad, end);
46  const auto start_mid = std::max(Nquad, beg);
47 
48  double Rint_ends = qip::inner_product_sub(beg, Nquad, cq, f1, rest...);
49 
50  const double Rint_mid =
51  qip::inner_product_sub(start_mid, end_mid, f1, rest...);
52 
53  for (auto i = end_mid; i < end; ++i) {
54  Rint_ends += cq[end_mid + Nquad - i - 1] * qip::multiply_at(i, f1, rest...);
55  }
56  return (Rint_mid + dq_inv * Rint_ends) * dt;
57 }

60 template <typename T>
61 inline std::vector<T> derivative(const std::vector<T> &f,
62  const std::vector<T> &drdt, const T dt,
63  const int order = 1)
64 // df/dr = df/dt * dt/dr = (df/dt) / (dr/dt)
65 // = (df/di) * (di/dt) / (dr/dt)
66 // = (df/di) / (dt * dr/dt)
67 // coeficients from: http://en.wikipedia.org/wiki/Finite_difference_coefficient
68 {
69  [[maybe_unused]] auto sp = IO::Profile::safeProfiler(__func__);
70 
71  std::size_t num_points = f.size();
72  std::vector<T> df(num_points);
73  if (num_points < 4)
74  return df;
75 
76  df[0] = (f[1] - f[0]) / (dt * drdt[0]);
77  df[num_points - 1] =
78  (f[num_points - 1] - f[num_points - 2]) / (dt * drdt[num_points - 1]);
79 
80  df[1] = (f[2] - f[0]) / (2 * dt * drdt[1]);
81  df[num_points - 2] =
82  (f[num_points - 1] - f[num_points - 3]) / (2 * dt * drdt[num_points - 2]);
83 
84  df[2] = (f[0] - 8 * f[1] + 8 * f[3] - f[4]) / (12 * dt * drdt[2]);
85  df[num_points - 3] = (f[num_points - 5] - 8 * f[num_points - 4] +
86  8 * f[num_points - 2] - f[num_points - 1]) /
87  (12 * dt * drdt[num_points - 3]);
88 
89  df[3] = (-1 * f[0] + 9 * f[1] - 45 * f[2] + 45 * f[4] - 9 * f[5] + 1 * f[6]) /
90  (60 * dt * drdt[3]);
91  df[num_points - 4] =
92  (-1 * f[num_points - 7] + 9 * f[num_points - 6] - 45 * f[num_points - 5] +
93  45 * f[num_points - 3] - 9 * f[num_points - 2] + 1 * f[num_points - 1]) /
94  (60 * dt * drdt[num_points - 4]);
95 
96  for (std::size_t i = 4; i < (num_points - 4); i++) {
97  df[i] = ((1.0 / 8) * f[i - 4] - (4.0 / 3) * f[i - 3] + 7 * f[i - 2] -
98  28 * f[i - 1] - (1.0 / 8) * f[i + 4] + (4.0 / 3) * f[i + 3] -
99  7 * f[i + 2] + 28 * f[i + 1]) /
100  (35 * dt * drdt[i]);
101  }
102 
103  if (order > 1)
104  df = derivative(df, drdt, dt, order - 1);
105 
106  return df;
107 }

111 template <Direction direction, typename Real>
112 inline void
113 additivePIntegral(std::vector<Real> &answer, const std::vector<Real> &f,
114  const std::vector<Real> &g, const std::vector<Real> &h,
115  const Grid &gr, std::size_t pinf = 0)
116 // answer(r) += f(r) * Int[g(r')*h(r') , {r',0,r}]
117 // Note '+=' - this is additive!
118 // Note: vector answer must ALREADY exist, and be correctly initialised!
119 // XXX later: be able to call with just f,g or just g ?
120 {
121  const auto size = g.size();
122  if (pinf == 0 || pinf >= size)
123  pinf = size;
124  const auto max = static_cast<int>(pinf - 1); // must be signed
125 
126  constexpr const bool forward = (direction == zero_to_r);
127  constexpr const int inc = forward ? +1 : -1;
128  const auto init = forward ? 0ul : std::size_t(max);
129  const auto fin = forward ? std::size_t(max) : 0ul;
130 
131  Real x = int(init) < max ? 0.5 * g[init] * h[init] * gr.drdu[init] : 0.0;
132  answer[init] += f[init] * x * gr.du;
133  for (auto i = int(init) + inc; i != int(fin) + inc; i += inc) {
134  const auto im = std::size_t(i - inc);
135  const auto i2 = std::size_t(i);
136  x += 0.5 * (g[im] * h[im] * gr.drdu[im] + g[i2] * h[i2] * gr.drdu[i2]);
137  answer[i2] += f[i2] * x * gr.du;
138  }
139  if (int(fin) < max)
140  answer[fin] += 0.5 * f[fin] * g[fin] * h[fin] * gr.drdu[fin] * gr.du;
141 }

171 inline double num_integrate(const std::function<double(double)> &f, double a,
172  double b, long unsigned n_pts,
173  t_grid type = linear) {
174  //
175 
176  const auto dt = (type == linear) ? (b - a) / double(n_pts - 1)
177  : std::log(b / a) / double(n_pts - 1);
178 
179  std::function<double(long unsigned)> x =
180  (type == linear) ? linx(a, dt) : logx(a, dt);
181  std::function<double(long unsigned)> dxdt =
182  (type == linear) ? one() : logx(a, dt);
183 
184  double Rint_s = 0.0;
185  for (long unsigned i = 0; i < Nquad; i++) {
186  Rint_s += cq[i] * f(x(i)) * dxdt(i);
187  }
188 
189  double Rint_m = 0.0;
190  for (auto i = Nquad; i < n_pts - Nquad; i++) {
191  Rint_m += f(x(i)) * dxdt(i);
192  }
193 
194  double Rint_e = 0;
195  for (auto i = n_pts - Nquad; i < n_pts; i++) {
196  Rint_e += cq[n_pts - i - 1] * f(x(i)) * dxdt(i);
197  }
198 
199  return (Rint_m + dq_inv * (Rint_s + Rint_e)) * dt;
200 }
